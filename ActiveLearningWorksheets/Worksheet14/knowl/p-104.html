\begin{equation*} \begin{aligned} \int_{0}^{1} 2x\sqrt{1-x^2} dx \amp =\int_{0}^{-1} -\sqrt{u}du\\ \amp =\int_{-1}^{0} \sqrt{u}du =\frac{2}{3} u^{\frac{3}{2}}\bigg|_{-1}^{0}\\ \amp =\frac{2}{3} \left(0\right)-\frac{2}{3} \left(-1\right)^{\frac{3}{2}}\\ \amp =\frac{2}{3}. \end{aligned} \end{equation*}
