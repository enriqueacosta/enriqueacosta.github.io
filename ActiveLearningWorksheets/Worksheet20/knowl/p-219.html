\begin{equation*} \begin{aligned} \int \int_{S} xz \, dS \amp =\int_{0}^{2} \int_{0}^{2-u} u(2-u-v) \sqrt{3} \,dv\,du\\ \amp =\sqrt{3} \int_{0}^{2} \int_{0}^{2-u} (2u-u^2-uv) \,dv\,du \\ \amp =\sqrt{3} \int_{0}^{2} \left( 2uv-u^2v-u\frac{v^2}{2} \right)\bigg|_{0}^{2-u} \,du \\ \amp =\sqrt{3} \int_{0}^{2} \left( \frac{u^3}{2} -2u^2+2u \right) \,du \\ \amp =\sqrt{3} \left( \frac{u^4}{8} -\frac{2u^2}{3}+u^2 \right)\bigg|_{0}^{2} \,du \\ \amp =\sqrt{3} \left( \frac{2^4}{8} -\frac{2(2)^2}{3}+2^2 \right)\bigg|_{0}^{2} \,du \\ \amp = \frac{2\sqrt{3}}{3}. \end{aligned} \end{equation*}
