\begin{equation*} \begin{aligned} x \amp =\amp \frac{-55\pm\sqrt{55^2-4({99} )({50} )}}{2({99} )}\\ x \amp =\amp \frac{-55\pm\sqrt{-{16775} }}{178} \end{aligned} \end{equation*}
