\documentclass[border=1pt, 12pt, tikz]{standalone}

\newcommand\wideOne{3cm}
\newcommand\wideTwo{4cm}
\newcommand\wideThree{2.7cm}
\newcommand\wideFour{3cm}
\newcommand\distOne{2cm}
\newcommand\distTwo{1cm}

\begin{document}
\begin{tikzpicture}[scale=1]

% 1st level
\draw[thick]%
   (0,0) 
   -- node[above, sloped, align=center]
      {infected} 
   (\wideOne,\distOne);
\draw[thick] 
   (0,0) 
   -- node[below, sloped, align=center]
      {not infected} 
   (\wideOne,-\distOne);

% 2nd, 3rd and 4th Level
\draw[thick] 
   (\wideOne,\distOne)
   -- node[above, align=center, sloped]
      {test says positive}
   ++ (\wideTwo,\distTwo) 
      node[right, align=center, text width=\wideThree]
      {true positive}
   ; 
\draw[thick] 
   (\wideOne,\distOne) 
   -- node[below, align=center, sloped]
      {test says negative}
   ++ (\wideTwo,-\distTwo) 
      node[right, align=center, text width=\wideThree]
      {false negative}
   ;
\draw[thick] 
   (\wideOne,-\distOne) 
   -- node[above, align=center, sloped]
      {test says positive}
   ++ (\wideTwo,\distTwo) 
      node[right, align=center, text width=\wideThree]
      {false positive}
   ;  
\draw[thick] 
   (\wideOne,-\distOne)
   -- node[below, align=center, sloped]
      {test says negative}
   ++ (\wideTwo,-\distTwo) 
      node[right, align=center, text width=\wideThree]
      {true negative}
   ; 
\end{tikzpicture}
\end{document}
