\documentclass[border=1pt, 12pt, tikz]{standalone}

\newcommand\wideOne{3cm}
\newcommand\wideTwo{3.5cm}
\newcommand\wideThree{2.6cm}
\newcommand\wideFour{3.2cm}
\newcommand\distOne{2cm}
\newcommand\distTwo{1cm}

\begin{document}
\begin{tikzpicture}[scale=1]

% 1st level
\draw[thick]%
   (0,0) 
   -- node[above, sloped, align=center]
      {heart} 
   (\wideOne,\distOne);
\draw[thick] 
   (0,0) 
   -- node[below, sloped, align=center]
      {not a heart} 
   (\wideOne,-\distOne);

% 2nd, 3rd and 4th Level
\draw[thick] 
   (\wideOne,\distOne)
   -- node[above, align=center, sloped]
      {divisible by 3}
   ++ (\wideTwo,\distTwo) 
      node[right, align=center, text width=\wideThree]
      {heart and\\divisible by 3}
   ; 
\draw[thick] 
   (\wideOne,\distOne) 
   -- node[below, align=center, sloped]
      {not divisible by 3}
   ++ (\wideTwo,-\distTwo) 
      node[right, align=center, text width=\wideThree]
      {heart and not divisible by 3}
   ;
\draw[thick] 
   (\wideOne,-\distOne) 
   -- node[above, align=center, sloped]
      {divisible by 3}
   ++ (\wideTwo,\distTwo) 
      node[right, align=center, text width=\wideThree]
      {not a heart and divisible by 3}
   ;  
\draw[thick] 
   (\wideOne,-\distOne)
   -- node[below, align=center, sloped]
      {not divisible by 3}
   ++ (\wideTwo,-\distTwo) 
      node[right, align=center, text width=\wideThree]
      {not a heart and not divisible by 3}
   ; 
\end{tikzpicture}
\end{document}
